75 /// `Jacobian(i,0)` is \f$\frac{1}{r}\partial x^i/\partial\theta\f$,
76 /// and `Jacobian(i,1)`
77 /// is \f$\frac{1}{r\sin\theta}\partial x^i/\partial\phi\f$.
78 /// Here \f$r\f$ means \f$\sqrt{x^2+y^2+z^2}\f$.
79 /// `Jacobian` doesn't depend on the shape of the surface.
80 template <typename Frame>

82  static constexpr db::DataBoxString label = "Jacobian";
83  static StrahlkorperTags_detail::Jacobian<Frame> function(
84  const db::item_type<ThetaPhi<Frame>>& theta_phi) noexcept;
85  using argument_tags = tmpl::list<ThetaPhi<Frame>>;
86 };

88 /// `InvJacobian(0,i)` is \f$r\partial\theta/\partial x^i\f$,
89 /// and `InvJacobian(1,i)` is \f$r\sin\theta\partial\phi/\partial x^i\f$.
90 /// Here \f$r\f$ means \f$\sqrt{x^2+y^2+z^2}\f$.
91 /// `InvJacobian` doesn't depend on the shape of the surface.

95  static StrahlkorperTags_detail::InvJacobian<Frame> function(
96  const db::item_type<ThetaPhi<Frame>>& theta_phi) noexcept;
97  using argument_tags = tmpl::list<ThetaPhi<Frame>>;
98 };

100 /// `InvHessian(k,i,j)` is \f$\partial (J^{-1}){}^k_j/\partial x^i\f$,
101 /// where \f$(J^{-1}){}^k_j\f$ is the inverse Jacobian.
102 /// `InvHessian` is not symmetric because the Jacobians are Pfaffian.
103 /// `InvHessian` doesn't depend on the shape of the surface.
104 template <typename Frame>

106  static constexpr db::DataBoxString label = "InvHessian";
107  static StrahlkorperTags_detail::InvHessian<Frame> function(
108  const db::item_type<ThetaPhi<Frame>>& theta_phi) noexcept;
109  using argument_tags = tmpl::list<ThetaPhi<Frame>>;
110 };

112 /// (Euclidean) distance \f$r(\theta,\phi)\f$ from the center to each
113 /// point of the surface.
114 template <typename Frame>
115 struct Radius : db::ComputeItemTag {
116  static constexpr db::DataBoxString label = "Radius";
117  SPECTRE_ALWAYS_INLINE static auto function(
118  const ::Strahlkorper<Frame>& strahlkorper) noexcept {
119  return strahlkorper.ylm_spherepack().spec_to_phys(
120  strahlkorper.coefficients());
121  }
122  using argument_tags = tmpl::list<Strahlkorper<Frame>>;
123 };

136 /// `DxRadius(i)` is \f$\partial r/\partial x^i\f$.
137 /// Here \f$r=r(\theta,\phi)\f$ is the surface, which is considered a

146  using argument_tags =
147  tmpl::list<Strahlkorper<Frame>, Radius<Frame>, InvJacobian<Frame>>;
148 };

178 /// Cartesian components of (unnormalized) one-form defining the surface.
179 /// This is computed by \f$x_i/r-\partial r/\partial x^i\f$,

187  const db::item_type<DxRadius<Frame>>& dx_radius,
188  const db::item_type<Rhat<Frame>>& r_hat) noexcept;
189  using argument_tags = tmpl::list<DxRadius<Frame>, Rhat<Frame>>;
190 };

192 /// `Tangents(j,i)` is \f$\partial x_{\rm surf}^i/\partial q^j\f$,

204  const db::item_type<Rhat<Frame>>& r_hat,
205  const db::item_type<Jacobian<Frame>>& jac) noexcept;
206  using argument_tags = tmpl::list<Strahlkorper<Frame>, Radius<Frame>,
207  Rhat<Frame>, Jacobian<Frame>>;
208 };
