27 /// \ingroup SurfacesGroup
28 /// \brief Computes 3-covariant gradient \f$D_i S_j\f$ of a
29 /// Strahlkorper's normal.
30 ///
31 /// \details See Eqs. (1--9) of https://arxiv.org/abs/gr-qc/9606010.
32 /// Here \f$S_j\f$ is the (normalized) unit one-form to the surface,
33 /// and \f$D_i\f$ is the spatial covariant derivative. Note that this
34 /// object is symmetric, even though this is not obvious from the
35 /// definition. The input arguments `r_hat`, `radius`, and
36 /// `d2x_radius` depend on the Strahlkorper but not on the metric, and
37 /// can be computed from a Strahlkorper using ComputeItems in
38 /// `StrahlkorperTags`. The input argument
39 /// `one_over_one_form_magnitude` is \f$1/\sqrt{g^{ij}n_i n_j}\f$,
40 /// where \f$n_i\f$ is `StrahlkorperTags::NormalOneForm` (i.e. the
41 /// unnormalized one-form to the Strahlkorper); it can be computed
42 /// using (one over) the `magnitude` function. The input argument
43 /// `unit_normal_one_form` is \f$S_j\f$,the normalized one-form.
44 template <typename Frame>
45 tnsr::ii<DataVector, 3, Frame> grad_unit_normal_one_form(
46  const tnsr::i<DataVector, 3, Frame>& r_hat, const DataVector& radius,
47  const tnsr::i<DataVector, 3, Frame>& unit_normal_one_form,
48  const tnsr::ii<DataVector, 3, Frame>& d2x_radius,
49  const DataVector& one_over_one_form_magnitude,
50  const tnsr::Ijj<DataVector, 3, Frame>& christoffel_2nd_kind) noexcept;

52 /// \ingroup SurfacesGroup
53 /// \brief Computes inverse 2-metric \f$g^{ij}-S^i S^j\f$ of a Strahlkorper.
54 ///
55 /// \details See Eqs. (1--9) of https://arxiv.org/abs/gr-qc/9606010.
56 /// Here \f$S^i\f$ is the (normalized) unit vector to the surface,
57 /// and \f$g^{ij}\f$ is the 3-metric. This object is expressed in the
58 /// usual 3-d Cartesian basis, so it is written as a 3-dimensional tensor.
59 /// But because it is orthogonal to \f$S_i\f$, it has only 3 independent
60 /// degrees of freedom, and could be expressed as a 2-d tensor with an
61 /// appropriate choice of basis. The input argument `unit_normal_vector` is
62 /// \f$S^i = g^{ij} S_j\f$, where \f$S_j\f$ is the unit normal one form.
63 template <typename Frame>
64 tnsr::II<DataVector, 3, Frame> inverse_surface_metric(
65  const tnsr::I<DataVector, 3, Frame>& unit_normal_vector,
66  const tnsr::II<DataVector, 3, Frame>& upper_spatial_metric) noexcept;

82 /*!
83  * \ingroup SurfacesGroup
84  * \brief Extrinsic curvature of a 2D `Strahlkorper` embedded in a 3D space.
85  *
86  * \details Implements Eq. (D.43) of Carroll's Spacetime and Geometry text.
87  * Specifically,
88  * \f$ K_{ij} = P^k_i P^l_j \nabla_{(k} S_{l)} \f$, where
89  * \f$ P^k_i = \delta^k_i - S^k S_i \f$,
90  * `grad_normal` is the quantity \f$ \nabla_k S_l \f$ returned by
91  * `StrahlkorperGr::grad_unit_normal_one_form`, and `unit_normal_vector` is
92  * \f$S^i = g^{ij} S_j\f$ where \f$S_j\f$ is the unit normal one form.
93  * Not to be confused with the extrinsic curvature of a 3D spatial slice
94  * embedded in 3+1 spacetime.
95  * Because StrahlkorperGr::grad_unit_normal_one_form is symmetric, this
96  * can be expanded into
97  * \f$ K_{ij} = \nabla_{i}S_{j} - 2 S^k S_{(i}\nabla_{j)}S_k
98  * + S_i S_j S^k S^l \nabla_{k} n_{l}\f$.
99  */
100 template <typename Frame>
101 tnsr::ii<DataVector, 3, Frame> extrinsic_curvature(
102  const tnsr::ii<DataVector, 3, Frame>& grad_normal,
103  const tnsr::i<DataVector, 3, Frame>& unit_normal_one_form,
104  const tnsr::I<DataVector, 3, Frame>& unit_normal_vector) noexcept;

106 /// \ingroup SurfacesGroup
107 /// \brief Intrinsic Ricci scalar of a 2D `Strahlkorper`.
108 ///
109 /// \details Implements Eq. (D.51) of
110 /// Sean Carroll's Spacetime and Geometry textbook (except correcting
111 /// sign errors: both extrinsic curvature terms are off by a minus sign
112 /// in Carroll's text but correct in Carroll's errata).
113 /// \f$ \hat{R}=R - 2 R_{ij} S^i S^j + K^2-K^{ij}K_{ij}.\f$
114 /// Here \f$\hat{R}\f$ is the intrinsic Ricci scalar curvature of
115 /// the Strahlkorper, \f$R\f$ and \f$R_{ij}\f$ are the Ricci scalar and
116 /// Ricci tensor of the 3D space that contains the Strahlkorper,
117 /// \f$ K_{ij} \f$ the output of StrahlkorperGr::extrinsic_curvature,
118 /// \f$ K \f$ is the trace of \f$K_{ij}\f$,
119 /// and `unit_normal_vector` is
120 /// \f$S^i = g^{ij} S_j\f$ where \f$S_j\f$ is the unit normal one form.
121 template <typename Frame>
122 Scalar<DataVector> ricci_scalar(
123  const tnsr::ii<DataVector, 3, Frame>& spatial_ricci_tensor,
124  const tnsr::I<DataVector, 3, Frame>& unit_normal_vector,
125  const tnsr::ii<DataVector, 3, Frame>& extrinsic_curvature,
126  const tnsr::II<DataVector, 3, Frame>& upper_spatial_metric) noexcept;
127 } // namespace StrahlkorperGr
