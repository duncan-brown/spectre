36 /// \ingroup ActionsGroup
37 /// \ingroup DiscontinuousGalerkinGroup
38 /// \brief Initialize a dG element with analytic initial data
39 ///
40 /// Uses:
41 /// - ConstGlobalCache:
42 /// * A tag deriving off of Cache::AnalyticSolutionBase
43 /// * CacheTags::TimeStepper
44 ///
45 /// DataBox changes:
46 /// - Adds:
47 /// * Tags::TimeId
48 /// * Tags::Time
49 /// * Tags::TimeStep
50 /// * Tags::LogicalCoordinates<Dim>
51 /// * Tags::Extents<Dim>
52 /// * Tags::Element<Dim>
53 /// * Tags::ElementMap<Dim>
54 /// * System::variables_tag
55 /// * Tags::HistoryEvolvedVariables<System::variables_tag,
56 /// db::add_tag_prefix<Tags::dt, System::variables_tag>>
57 /// * Tags::Coordinates<Tags::ElementMap<Dim>,
58 /// Tags::LogicalCoordinates<Dim>>
59 /// * Tags::InverseJacobian<Tags::ElementMap<Dim>,
60 /// Tags::LogicalCoordinates<Dim>>
61 /// * Tags::deriv<System::variables_tag::tags_list,
62 /// System::gradients_tags,
63 /// Tags::InverseJacobian<Tags::ElementMap<Dim>,
64 /// Tags::LogicalCoordinates<Dim>>>
65 /// * db::add_tag_prefix<Tags::dt, System::variables_tag>
66 /// * Tags::UnnormalizedFaceNormal<Dim>
67 /// - Removes: nothing
68 /// - Modifies: nothing
69 template <size_t Dim>
70 struct InitializeElement {
71  template <typename Tag>
72  using interface_tag = Tags::Interface<Tags::InternalDirections<Dim>, Tag>;
73 
74  template <class System>
75  using return_tag_list = tmpl::list<
76  Tags::TimeId, Tags::Time, Tags::TimeStep, Tags::LogicalCoordinates<Dim>,
77  Tags::Extents<Dim>, Tags::Element<Dim>, Tags::ElementMap<Dim>,
78  typename System::variables_tag,
79  Tags::HistoryEvolvedVariables<
80  typename System::variables_tag,
81  db::add_tag_prefix<Tags::dt, typename System::variables_tag>>,
82  Tags::Coordinates<Tags::ElementMap<Dim>, Tags::LogicalCoordinates<Dim>>,
83  Tags::InverseJacobian<Tags::ElementMap<Dim>,
84  Tags::LogicalCoordinates<Dim>>,
85  Tags::deriv<typename System::variables_tag::tags_list,
86  typename System::gradients_tags,
87  Tags::InverseJacobian<Tags::ElementMap<Dim>,
88  Tags::LogicalCoordinates<Dim>>>,
89  db::add_tag_prefix<Tags::dt, typename System::variables_tag>,
90  Tags::InternalDirections<Dim>,
91  interface_tag<Tags::Direction<Dim>>,
92  interface_tag<Tags::Extents<Dim - 1>>,
93  interface_tag<Tags::UnnormalizedFaceNormal<Dim>>,
94  interface_tag<typename System::template magnitude_tag<
95  Tags::UnnormalizedFaceNormal<Dim>>>,
96  interface_tag<Tags::Normalized<Tags::UnnormalizedFaceNormal<Dim>,
97  typename System::template magnitude_tag<
98  Tags::UnnormalizedFaceNormal<Dim>>>>,
99  // This should be the boundary forms of
100  // System::normal_dot_fluxes::argument_tags, but it is not clear
101  // how to get that. System::variables_tag should generally be a
102  // superset.
103  interface_tag<typename System::variables_tag>>;
104 
105  template <typename... InboxTags, typename Metavariables, typename ActionList,
106  typename ParallelComponent>
107  static auto apply(const db::DataBox<tmpl::list<>>& /*box*/,
108  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
109  const Parallel::ConstGlobalCache<Metavariables>& cache,
110  const ElementIndex<Dim>& array_index,
111  const ActionList /*meta*/,
112  const ParallelComponent* const /*meta*/,
113  std::vector<std::array<size_t, Dim>> initial_extents,
114  Domain<Dim, Frame::Inertial> domain, Time initial_time,
115  TimeDelta initial_dt) noexcept {
116  using solution_tag = CacheTags::AnalyticSolutionBase;
117  using variables_tags =
118  typename Metavariables::system::variables_tag::tags_list;
119 
120  ElementId<Dim> element_id{array_index};
121 
122  const auto& my_block = domain.blocks()[element_id.block_id()];
123 
124  ElementMap<Dim, Frame::Inertial> map{element_id,
125  my_block.coordinate_map().get_clone()};
126 
127  Element<Dim> element = create_initial_element(element_id, my_block);
128 
129  ::Index<Dim> mesh{initial_extents[element_id.block_id()]};
130  const auto num_grid_points = mesh.product();
131  auto logical_coords = logical_coordinates(mesh);
132  auto inertial_coords = map(logical_coords);
133 
134  // Set initial data from analytic solution
135  Variables<variables_tags> vars{num_grid_points};
136  vars.assign_subset(Parallel::get<solution_tag>(cache).variables(
137  inertial_coords, initial_time.value(), variables_tags{}));
138 
139  typename Tags::HistoryEvolvedVariables<
140  Tags::Variables<variables_tags>,
141  Tags::dt<Tags::Variables<db::wrap_tags_in<Tags::dt, variables_tags>>>>::
142  type history_dt_vars;
143  const auto& time_stepper = Parallel::get<CacheTags::TimeStepper>(cache);
144  if (not time_stepper.is_self_starting()) {
145  // We currently just put initial points at past slab boundaries.
146  Time past_t = initial_time;
147  TimeDelta past_dt = initial_dt;
148  for (size_t i = time_stepper.number_of_past_steps(); i > 0; --i) {
149  past_dt = past_dt.with_slab(past_dt.slab().advance_towards(-past_dt));
150  past_t -= past_dt;
151  Variables<variables_tags> hist_vars{num_grid_points};
152  Variables<db::wrap_tags_in<Tags::dt, variables_tags>> dt_hist_vars{
153  num_grid_points};
154  hist_vars.assign_subset(Parallel::get<solution_tag>(cache).variables(
155  inertial_coords, past_t.value(), variables_tags{}));
156  dt_hist_vars.assign_subset(Parallel::get<solution_tag>(cache).variables(
157  inertial_coords, past_t.value(),
158  db::wrap_tags_in<Tags::dt, variables_tags>{}));
159  history_dt_vars.insert_initial(past_t, std::move(hist_vars),
160  std::move(dt_hist_vars));
161  }
162  }
163 
164  // Set up initial time
165  TimeId time_id{};
166  time_id.time = initial_time;
167 
168  db::compute_databox_type<return_tag_list<typename Metavariables::system>>
169  outbox = db::create<
170  db::get_items<return_tag_list<typename Metavariables::system>>,
171  db::get_compute_items<
172  return_tag_list<typename Metavariables::system>>>(
173  time_id, initial_dt, std::move(mesh), std::move(element),
174  std::move(map), std::move(vars), std::move(history_dt_vars),
175  Variables<db::wrap_tags_in<Tags::dt, variables_tags>>{
176  num_grid_points, 0.0});
177 
178  return std::make_tuple(std::move(outbox));
179  }
180 };
