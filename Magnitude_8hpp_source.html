12 /*!
13  * \ingroup TensorGroup
14  * \brief Compute the Euclidean magnitude of a rank-1 tensor
15  *
16  * \details
17  * Computes the square root of the sum of the squares of the components of
18  * the rank-1 tensor.
19  */
20 template <typename DataType, typename Index>
21 Scalar<DataType> magnitude(
22  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector) noexcept {
23  return Scalar<DataType>{sqrt(get(dot_product(vector, vector)))};
24 }

26 /*!
27  * \ingroup TensorGroup
28  * \brief Compute the magnitude of a rank-1 tensor
29  *
30  * \details
31  * Returns the square root of the input tensor contracted twice with the given
32  * metric.
33  */
34 template <typename DataType, typename Index>
35 Scalar<DataType> magnitude(
36  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector,
37  const Tensor<DataType, Symmetry<1, 1>,
38  index_list<change_index_up_lo<Index>,
39  change_index_up_lo<Index>>>&
40  metric) noexcept {
41  return Scalar<DataType>{sqrt(get(dot_product(vector, vector, metric)))};
42 }
